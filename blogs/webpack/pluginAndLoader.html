    # 一个压缩插件

    插件的基本组成：插件名称， apply方法，插件的hooks, 其他处理逻辑

    zip-plugin

    准备：webpack资源写入 (opens new window) jszip (opens new window)

    const JSZip = require('jszip');
const path = require('path');
const RawSource = require('webpack-sources').RawSource;
const zip = new JSZip();
class ZipWebpackPlugin {
    constructor(options) {
      // console.log(options)
      this.options =options
    }
    apply(compiler) {
      const self =this;
      //hooks.emit 定义在某个时刻
      compiler.hooks.emit.tapAsync("ZipWebpackPlugin", (compilation, cb) => {
        const folder  = zip.folder(this.options.filename);
        for(let filename in compilation.assets) {
         const source =compilation.assets[filename].source();
         folder.file(filename, source)
        }
        zip.generateAsync({
          type: 'nodebuffer'
        }).then((content) => {
          
          const outputPath = path.join(compilation.options.output.path, this.options.filename + '.zip');
          const outputPathRelative = path.relative(compilation.options.output.path, outputPath);
          console.log(outputPathRelative)
          compilation.assets[outputPathRelative] = new RawSource(content);
          cb();
        });
        
      });
    }
  };
  module.exports = ZipWebpackPlugin;

    # webpack 自动合并雪碧图loader

    命名： sprite-picture-loader

    目标： 将多张图片合成为一张图片

    支持语法：

    • background: url('a.png?__sprite');
									-->  background: url('sprite.png');
										 background-position: 0 100px;
background: url('b.png?__sprite');

      2
      3
      4

    准备： 使用spritesmith (opens new window)合成图片

    ​ 使用loader-runner (opens new window) 在不安装webpack的情况下运行loader,进行loader的开发和调试。

    const fs = require('fs');
const { getOptions } = require("loader-utils");
const Spritesmith = require('spritesmith');
const path = require('path');
const regg = /url\(\S?(\S*)\?__sprite\S?\)/g;
const reg = /url\(\S?(\S*)\?__sprite\S?\)/;
function existsSync(path) {
    return fs.existsSync(path)
}
function mkdirSync(path) {
    return fs.mkdirSync(path)
}

module.exports = function (source) {
    const callback = this.async();
    // 可配置 path , filename, 及图片类型 extension
    const defalutOptions = {
        path: './dist/',
        extension: 'png',
        filename: 'sprite',
        width: true,
        height: true,
        backgroundPosition: true,
        backgroundRepeat: true
    }
    let loaderOptions = getOptions(this) || {};
    loaderOptions = Object.assign(loaderOptions, defalutOptions);
    const { filename, extension, path: pathUrl } = loaderOptions;
	// 匹配url('*******')
    const imgs = source.match(regg);
    const matchImgs = [];
    for(let i = 0; i < imgs.length; i++) {
        const img = imgs[i].match(reg)[1];
        // 防止重复添加
        if(matchImgs.indexOf(path.resolve(this.context, img)) === -1) {
            matchImgs.push(path.resolve(this.context, img));
        }
    }
    
    Spritesmith.run({src: matchImgs},(err, result) => {
        if(err) {
            throw new Error('Sprite:', err)
        }
        // 讲图片写入文件夹中
        const url = path.resolve(__dirname, `${pathUrl}`);
        const coordinates = result.coordinates;
        if(!fs.existsSync(url)) {
            fs.mkdirSync(url)
        }
        fs.writeFileSync(`${url}/${filename}.${extension}`, result.image);
		// 替换css中的url，添加其他的一些属性，存在的问题，如果当前为backgound:url(./sprite.jpg) 后面接了其他值，会被删除。
        source = source.replace(regg, (match) => {
            let pathName = path.resolve(this.context, match.match(reg)[1]);
            let {x, y, width, height} = coordinates[pathName];

            // 模板字符串格式换行缩进问题，注意换行
            return `url(./sprite.jpg);
    background-position: ${ -x }px ${ -y }px;
    background-repeat:no-repeat;
    width: ${width}px;
    height: ${height}px`
        });
		// 将css写入到文件夹中
        fs.writeFileSync(path.resolve(__dirname, `${pathUrl}/${filename}.css`), source);
        callback(null, source)
    });
}
